8pub fn lang_token(
9    ctx: &ParseContext,
10    text: &str,
11    pos: NodeSpan,
12    token_kind: types::Token,
13) -> Result<(LexerPosition, Token, LexerPosition), (types::Token, types::token::ErrorKind)> {
14    Ok((
15        pos.start(),
16        match token_kind {
17            types::Token::IDENT(ident) => {
18                if ctx.is_type_name(&ident) {
19                    Token::TypeName(ident)
20                } else {
21                    // It is an identifier
22                    Token::Identifier(ident)
23                }
24            }
25            types::Token::TYPE_NAME(type_name) => match type_name {
26                types::TypeName::VOID => Token::Void,
27                types::TypeName::INT => Token::Int,
28                types::TypeName::BOOL => Token::Bool,
29                types::TypeName::FLOAT => Token::Float,
30                types::TypeName::DOUBLE => Token::Double,
31                // HLSL vector types (enum variants use GLSL-style names internally)
32                types::TypeName::VEC2 => Token::Vec2,
33                types::TypeName::VEC3 => Token::Vec3,
34                types::TypeName::VEC4 => Token::Vec4,
35                types::TypeName::IVEC2 => Token::IVec2,
36                types::TypeName::IVEC3 => Token::IVec3,
37                types::TypeName::IVEC4 => Token::IVec4,
38                types::TypeName::BVEC2 => Token::BVec2,
39                types::TypeName::BVEC3 => Token::BVec3,
40                types::TypeName::BVEC4 => Token::BVec4,
41                types::TypeName::UINT => Token::UInt,
42                types::TypeName::UVEC2 => Token::UVec2,
43                types::TypeName::UVEC3 => Token::UVec3,
44                types::TypeName::UVEC4 => Token::UVec4,
45                // HLSL matrix types (enum variants use GLSL-style names internally)
46                types::TypeName::MAT2 => Token::Mat2,
47                types::TypeName::MAT3 => Token::Mat3,
48                types::TypeName::MAT4 => Token::Mat4,
49                types::TypeName::MAT2X2 => Token::Mat22,
50                types::TypeName::MAT2X3 => Token::Mat2x3,
51                types::TypeName::MAT2X4 => Token::Mat2x4,
52                types::TypeName::MAT3X2 => Token::Mat3x2,
53                types::TypeName::MAT3X3 => Token::Mat33,
54                types::TypeName::MAT3X4 => Token::Mat3x4,
55                types::TypeName::MAT4X2 => Token::Mat4x2,
56                types::TypeName::MAT4X3 => Token::Mat4x3,
57                types::TypeName::MAT4X4 => Token::Mat44,
58                types::TypeName::MAT1X1 => Token::Mat22, // Map to closest equivalent
59                types::TypeName::MAT1X2 => Token::Mat2x3, // Map to closest equivalent
60                types::TypeName::MAT1X3 => Token::Mat2x3, // Map to closest equivalent
61                types::TypeName::MAT1X4 => Token::Mat2x4, // Map to closest equivalent
62                types::TypeName::MAT2X1 => Token::Mat3x2, // Map to closest equivalent
63                types::TypeName::MAT3X1 => Token::Mat3x2, // Map to closest equivalent
64                types::TypeName::MAT4X1 => Token::Mat4x2, // Map to closest equivalent
65                // Basic HLSL sampler type
66                types::TypeName::SAMPLER => Token::Sampler,
67                // HLSL texture types
68                types::TypeName::TEXTURE_1D => Token::Texture1D,
69                types::TypeName::TEXTURE_2D => Token::Texture2D,
70                types::TypeName::TEXTURE_3D => Token::Texture3D,
71                types::TypeName::TEXTURE_CUBE => Token::TextureCube,
72                types::TypeName::TEXTURE_1D_ARRAY => Token::Texture1DArray,
73                types::TypeName::TEXTURE_2D_ARRAY => Token::Texture2DArray,
74                types::TypeName::TEXTURE_CUBE_ARRAY => Token::TextureCubeArray,
75                types::TypeName::TEXTURE_2D_MS => Token::Texture2DMS,
76                types::TypeName::TEXTURE_2D_MS_ARRAY => Token::Texture2DMSArray,
77                types::TypeName::TEXTURE_2D_RECT => Token::Texture2DRect,
78                types::TypeName::TEXTURE_BUFFER => Token::TextureBuffer,
79                other => Token::TypeName(other.to_string().into()),
80            },
81            types::Token::FLOAT_CONST(val) => Token::FloatConstant(val),
82            types::Token::INT_CONST(val) => Token::IntConstant(val),
83            types::Token::UINT_CONST(val) => Token::UIntConstant(val),
84            types::Token::BOOL_CONST(val) => Token::BoolConstant(val),
85            types::Token::DOUBLE_CONST(val) => Token::DoubleConstant(val),
86            types::Token::STRING_CONST(val) => Token::StringConstant(val),
87            types::Token::LEFT_OP => Token::LeftOp,
88            types::Token::RIGHT_OP => Token::RightOp,
89            types::Token::INC_OP => Token::IncOp,
90            types::Token::DEC_OP => Token::DecOp,
91            types::Token::LE_OP => Token::LeOp,
92            types::Token::GE_OP => Token::GeOp,
93            types::Token::EQ_OP => Token::EqOp,
94            types::Token::NE_OP => Token::NeOp,
95            types::Token::AND_OP => Token::AndOp,
96            types::Token::OR_OP => Token::OrOp,
97            types::Token::XOR_OP => Token::XorOp,
98            types::Token::MUL_ASSIGN => Token::MulAssign,
99            types::Token::DIV_ASSIGN => Token::DivAssign,
100            types::Token::ADD_ASSIGN => Token::AddAssign,
101            types::Token::MOD_ASSIGN => Token::ModAssign,
102            types::Token::LEFT_ASSIGN => Token::LeftAssign,
103            types::Token::RIGHT_ASSIGN => Token::RightAssign,
104            types::Token::AND_ASSIGN => Token::AndAssign,
105            types::Token::XOR_ASSIGN => Token::XorAssign,
106            types::Token::OR_ASSIGN => Token::OrAssign,
107            types::Token::SUB_ASSIGN => Token::SubAssign,
108            types::Token::LPAREN => Token::LeftParen,
109            types::Token::RPAREN => Token::RightParen,
110            types::Token::LBRACKET => Token::LeftBracket,
111            types::Token::RBRACKET => Token::RightBracket,
112            types::Token::LBRACE => Token::LeftBrace,
113            types::Token::RBRACE => Token::RightBrace,
114            types::Token::PERIOD => Token::Dot,
115            types::Token::COMMA => Token::Comma,
116            types::Token::COLON => Token::Colon,
117            types::Token::EQUAL => Token::Equal,
118            types::Token::SEMICOLON => Token::Semicolon,
119            types::Token::BANG => Token::Bang,
120            types::Token::DASH => Token::Dash,
121            types::Token::TILDE => Token::Tilde,
122            types::Token::PLUS => Token::Plus,
123            types::Token::ASTERISK => Token::Star,
124            types::Token::SLASH => Token::Slash,
125            types::Token::PERCENT => Token::Percent,
126            types::Token::LANGLE => Token::LeftAngle,
127            types::Token::RANGLE => Token::RightAngle,
128            types::Token::BAR => Token::VerticalBar,
129            types::Token::CARET => Token::Caret,
130            types::Token::AMPERSAND => Token::Ampersand,
131            types::Token::QUESTION => Token::Question,
132            types::Token::HASH => {
133                return Err((token_kind, types::token::ErrorKind::InvalidToken));
134            }
135            types::Token::CONST => Token::Const,
136            types::Token::UNIFORM => Token::Uniform,
137            types::Token::BUFFER => Token::Buffer,
138            types::Token::SHARED => Token::Shared,
139            types::Token::COHERENT => Token::Coherent,
140            types::Token::VOLATILE => Token::Volatile,
141            types::Token::RESTRICT => Token::Restrict,
142            types::Token::READONLY => Token::ReadOnly,
143            types::Token::WRITEONLY => Token::WriteOnly,
144            types::Token::REGISTER => Token::Register,
145            types::Token::LAYOUT => Token::Layout,
146            types::Token::CENTROID => Token::Centroid,
147            types::Token::FLAT => Token::Flat,
148            types::Token::SMOOTH => Token::Smooth,
149            types::Token::NOPERSPECTIVE => Token::NoPerspective,
150            types::Token::PATCH => Token::Patch,
151            types::Token::SAMPLE => Token::Sample,
152            types::Token::INVARIANT => Token::Invariant,
153            types::Token::PRECISE => Token::Precise,
154            types::Token::BREAK => Token::Break,
155            types::Token::CONTINUE => Token::Continue,
156            types::Token::DO => Token::Do,
157            types::Token::FOR => Token::For,
158            types::Token::WHILE => Token::While,
159            types::Token::SWITCH => Token::Switch,
160            types::Token::CASE => Token::Case,
161            types::Token::DEFAULT => Token::Default,
162            types::Token::IF => Token::If,
163            types::Token::ELSE => Token::Else,
164            types::Token::SUBROUTINE => Token::Subroutine,
165            types::Token::IN => Token::In,
166            types::Token::OUT => Token::Out,
167            types::Token::INOUT => Token::InOut,
168            types::Token::DISCARD => Token::Discard,
169            types::Token::RETURN => Token::Return,
170            types::Token::LOWP => Token::LowPrecision,
171            types::Token::MEDIUMP => Token::MediumPrecision,
172            types::Token::HIGHP => Token::HighPrecision,
173            types::Token::PRECISION => Token::Precision,
174            types::Token::STRUCT => Token::Struct,
175            types::Token::CBUFFER => Token::CBuffer,
176            types::Token::SAMPLER_STATE => Token::SamplerState,
177            types::Token::SAMPLER_COMPARISON_STATE => Token::SamplerComparisonState,
178            types::Token::ATTRIBUTE => Token::Attribute,
179            types::Token::VARYING => Token::Varying,
180            types::Token::COMMON
181            | types::Token::PARTITION
182            | types::Token::ACTIVE
183            | types::Token::ASM
184            | types::Token::CLASS
185            | types::Token::UNION
186            | types::Token::ENUM
187            | types::Token::TYPEDEF
188            | types::Token::TEMPLATE
189            | types::Token::THIS
190            | types::Token::RESOURCE
191            | types::Token::GOTO
192            | types::Token::INLINE
193            | types::Token::NOINLINE
194            | types::Token::PUBLIC
195            | types::Token::STATIC
196            | types::Token::EXTERN
197            | types::Token::EXTERNAL
198            | types::Token::INTERFACE
199            | types::Token::LONG
200            | types::Token::SHORT
201            | types::Token::HALF
202            | types::Token::FIXED
203            | types::Token::UNSIGNED
204            | types::Token::SUPERP
205            | types::Token::OUTPUT
206            | types::Token::FILTER
207            | types::Token::SIZEOF
208            | types::Token::CAST
209            | types::Token::NAMESPACE
210            | types::Token::USING => {
211                return Err((token_kind, types::token::ErrorKind::InvalidToken));
212            }
213            types::Token::WS => Token::Whitespace,
214            types::Token::COMMENT => {
215                if text.starts_with("//") {
216                    Token::SingleLineComment
217                } else {
218                    Token::MultiLineComment
219                }
220            }
221            types::Token::ERROR(kind) => {
222                return Err((token_kind, kind));
223            }
224        },
225        pos.end(),
226    ))
227}
